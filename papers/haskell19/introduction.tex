data Html
  =  Text  String
  |  Sing  String
  |  Tag   String Html
  |  Html  :+: Html
\end{code}

\begin{code}
br :: Html
br = Sing "br"

bold :: Html -> Html
bold = Tag "b"

(<+>) :: Html -> Html -> Html
(<+>) x y = x :+: br :+: y

list :: [Html] -> Html
list xs = Tag "ul" (foldl1 (:+:) (Tag "li" <$> xs))

simplify :: Html -> Html
simplify (Text t1 :+: Text t2)
  = Text (t1 ++ t2)
simplify ((Text t1 :+: x) :+: y)
  = simplify (Text t1 :+: simplify (x :+: y))
simplify (x :+: y)
  = simplify x :+: simplify y
simplify (Tag t x)
  = Tag t (simplify x)
simplify x = x
\end{code}
